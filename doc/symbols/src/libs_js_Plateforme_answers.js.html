1 /**
  2  * Constructor of the Answers class
  3  *
  4  * @classdesc This class allows the simulator to answer a set of questions, and also make votes on the questions
  5  * @class     Answers
  6  * 
  7  */
  8 TestsCoco.Simulator.Answers = function(){
  9     this.dayInMillisecond = 86400000;
 10     this.hourInMillisecond = 3600000;
 11     this.minuteInMillisecond = 60000;
 12     this.answer_rate = 0.7;
 13 };
 14 
 15 /**
 16  * Generate the timecode of the answer
 17  *
 18  * @method    generateTime
 19  * @param      {Date}  session_start  
 20  * @param      {Object}  question       The question to answer
 21  * @param      {String}  profile        The user profile
 22  *
 23  * @return     {Date}    The date of the answer
 24  */
 25 TestsCoco.Simulator.Answers.prototype.generateTime = function (session_start,question,profile){
 26     var d = session_start.getTime(), d2;
 27     switch(profile) {
 28         case "regular":
 29             d += question.begin;
 30             break;
 31         case "random":
 32             d += _.random(question.begin,question.end);
 33             break;
 34         default:
 35             console.log('Profile not implemented');
 36     }
 37     d2 = new Date(d);
 38     return d2.toISOString();
 39 }

 41 /**
 42  * Generate the answer of the user
 43  *
 44  * @method    generateAnswer
 45  * @param      {Object}  question       The question to answer
 46  * @param      {Object}  user           The user who answer
 47  * @param      {Date}    session_start  
 48  * @param      {number}  session_id     
 49  *
 50  * @return     {Object}  The answer of the user
 51  */
 52 TestsCoco.Simulator.Answers.prototype.generateAnswer = function (question,user,session_start,session_id){
 53     var ret = {};
 54     ret.username = user.name;
 55     ret.subject = question.id;
 56     ret.date = this.generateTime(session_start,question,user.profile);
 57     ret.sessionId = session_id;
 58     if(Math.random() < user.bias.answer_rate){
 59         if(Math.random() < user.bias.right_answer_rate){
 60             ret.property = "right_answer";
 61             ret.value = question.content.answers.indexOf(question.content.answers.filter(function(a){return a.correct})[0])+1;
 62         }else{
 63             ret.property = "wrong_answer";
 64             ret.value = _.sample(question.content.answers.map(function(value,index){
 65                                 return (value.correct) ? -1 : index;
 66                             }).filter(function(i){return i>-1}));
 67         }
 68        
 69     }else{
 70         ret.property = "skipped_answer";
 71         ret.value = 0;
 72     }
 73     return ret;
 74 }

 76 /**
 77  * Generate the vote of the user
 78  *
 79  * @method    generateVote
 80  * @param      {Object}  question       The question to answer
 81  * @param      {Object}  user           The user who answer
 82  * @param      {Date}    session_start  
 83  * @param      {number}  session_id     
 84  *
 85  * @return     {Array}   The vote and the answer of the user
 86  */
 87 TestsCoco.Simulator.Answers.prototype.generateVote = function (question,user,session_start,session_id){
 88     var retour;
 89     var vote = {};
 90     vote.username = user.name;
 91     vote.subject = question.id;
 92     vote.date = this.generateTime(session_start,question,user.profile);
 93     vote.sessionId = session_id;
 94     
 95     var ans = this.generateAnswer(question,user,session_start,session_id);
 96     if(ans.property != "skipped_answer"){
 97         if(Math.random() < user.bias.vote_rate){
 98             if(Math.random() < user.bias.usefull_vote_rate){
 99                 vote.property = "usefull";
100                 vote.value = 1;
101             }else{
102                 vote.property = "useless";
103                 vote.value = -1
104             }
105         }else{
106             vote.property = "skipped_vote";
107             vote.value = 0;
108         }
109         retour = [ans,vote];
110     }else{
111         retour = [ans];
112     }
113     
114     return retour;
115 }

117 /**
118  * Generate the answers for the set of questions
119  *
120  * @method    generate
121  * @param      {Object}  question       The question to answer
122  * @param      {Object}  user           The user who answer
123  * @param      {Date}    session_start  
124  * @param      {number}  session_id     
125  *
126  * @return     {Array}   All the answers of the user
127  */
128 TestsCoco.Simulator.Answers.prototype.generate = function (questions,user,session_start,session_id){
129     var _this = this;
130     var reponses = [];
131 
132     $.each(questions,function(index,value){
133         reponses = reponses.concat(_this.generateVote(value,user,session_start,session_id));
134     });
135     return reponses;
136 }

141  * @method    main

150     var ret = [];
151 
152     ret = ret.concat(_this.generate(questions.annotations,user,session.date,session.id));
153     return ret;
