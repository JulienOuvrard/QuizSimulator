 15 /**
 16  * Count the occurences of many properties
 17  *
 18  * @method    countOccurences
 19  * @param      {Object[]}  tab_ans    The array of answer to analyse
 20  * @param      {Object[]}  tab_quest  The array of questions to analyse
 21  *
 22  * @return     {Object}  The properties 
 23  */
 24 TestsCoco.Simulator.Chooser.prototype.countOccurences = function (tab_ans,tab_quest){
 25  
 26     var ann = tab_quest.annotations;
 27     
 28     var shown_properties = ["right_answer","wrong_answer","skipped_answer"];
 29     var votted_properties = ["usefull","useless","skipped_vote"];
 30     
 31     var shown = {};
 32     var votted = {};
 33     var positive_vote = {};
 34     
 35     tab_ans.forEach(function(elem){
 36         if(jQuery.inArray(elem.property,shown_properties) != -1){
 37             if(elem.subject in shown){
 38                 shown[elem.subject] = ++shown[elem.subject];
 39             }
 40             else{
 41                 shown[elem.subject] = 1;
 42             }
 43         }
 44         if(jQuery.inArray(elem.property,votted_properties) != -1){
 45             if(elem.subject in votted){
 46                 votted[elem.subject] = ++votted[elem.subject];
 47             }
 48             else{
 49                 votted[elem.subject] = 1;
 50             }
 51         }
 52         if(elem.property=="usefull"){
 53             if(elem.subject in positive_vote){
 54                 positive_vote[elem.subject] = ++positive_vote[elem.subject];
 55             }
 56             else{
 57                 positive_vote[elem.subject] = 1;
 58             }
 59         }
 60     });
 61     
 62     tab_quest.forEach(function(elem){
 63         if(jQuery.inArray(elem.id,Object.keys(shown)) == -1){
 64             shown[elem.id] = 0;
 65         }
 66         if(jQuery.inArray(elem.id,Object.keys(votted)) == -1){
 67             votted[elem.id] = 0;
 68         }
 69         if(jQuery.inArray(elem.id,Object.keys(positive_vote)) == -1){
 70             positive_vote[elem.id] = 0;
 71         }
 72     });
 73     
 74     return {"shown":shown,"votted":votted,"positive_vote":positive_vote};
 75 }

 77 /**
 78  * Get the percentage of value of <tt>tab1</tt> by the value of <tt>tab2</tt>
 79  *
 80  * @method    percentage
 81  * @param      {Object}                tab1    The first Object
 82  * @param      {Object}                tab2    The second Object
 83  *
 84  * @return     {Object}  The object of percentages
 85  */
 86 TestsCoco.Simulator.Chooser.prototype.percentage = function (tab1,tab2){
 87     var result = {};
 88     $.each(tab2, function(index, value) {
 89         if(!tab1[index]){
 90             result[index]=0;
 91         }else{
 92             result[index]=tab1[index] / value;
 93         }
 94     });
 95     return result;
 96 }

135 /**

140  * @return     {Object}  The votes count
141  */
142 TestsCoco.Simulator.Chooser.prototype.positive = function (tab){
143     var votted_properties = ["usefull","useless","skipped_vote"];
144     var res={};
145     tab.forEach(function(elem){
146         if($.inArray(elem.property,votted_properties) != -1){

150                  res[elem.subject] = 0;
151              }
152         }
153     });

157 /**
158  * Compute the syntax simimarity between two sentences using the natural library
159  *
160  * @method     syntax_similarity
161  * @param      {String}              phrase1  The first sentence
162  * @param      {String}              phrase2  The second sentence
163  * @return     {number}  The syntax similarity score
164  */
165 TestsCoco.Simulator.Chooser.prototype.syntax_similarity = function (phrase1,phrase2){
166     var tool =  new TestsCoco.Tools();
167     natural.PorterStemmerFr.attach();
168     var p1=phrase1.tokenizeAndStem(),
169         p2=phrase2.tokenizeAndStem();
170 
171     
172     if(p1.length == 0 || p2.length == 0)
173         return 0;
174 
175     var words=_.uniq(p1.concat(p2));
176 
177 
178     var TfIdf = natural.TfIdf, tfidf = new TfIdf();
179     tfidf.addDocument(p1);
180     tfidf.addDocument(p2);
181     var row =[];
182     
183     var j = 0;
184     words.forEach(function(elem){
185         var temp = [];
186         tfidf.tfidfs(elem, function(i, measure) {
187             temp[i]=measure;
188         });
189         row[j]=temp;
190         ++j;
191     })
192     
193     var column = tool.transpose(row);
194 
195     return tool.cosine(column[0],column[1]);
196 }

198 /**
199  * Calculate the time similarity between two questions
200  *
201  * @method     time_similarity
202  * @param      {number}  t_q1      The time of the first question
203  * @param      {number}  t_q2      The time of the second question
204  * @param      {number}  max_time  The max time of all questions
205  * @return     {number}  The time similarity score
206  */
207 TestsCoco.Simulator.Chooser.prototype.time_similarity = function (t_q1,t_q2,max_time){
208     return 1 - (Math.abs((t_q1 - t_q2)) / max_time);
209 }

211 /**
212  * Compute the similarity between two questions
213  *
214  * @method     QuestionSimilarity
215  * @param      {Object}  q1      The first question
216  * @param      {Object}  q2      The second question
217  * @return     {number}  The similarity score
218  */
219 TestsCoco.Simulator.Chooser.prototype.QuestionSimilarity = function (q1,q2){
220     var tool = new TestsCoco.Tools();
221     var max_time = tool.getMaxOfArray(tool.getValuesOfObject(this.time));
222     var t_sim = this.time_similarity(this.time[q1],this.time[q2],max_time);
223     var s_sim = this.syntax_similarity(this.enonces[q1],this.enonces[q2]);
224     return t_sim * s_sim;
225 }

227 /**
228  * Compute all the similarity of all questions
229  *
230  * @method     similarity
231  * @param      {Object}  tab     The questions
232  * @return     {Array}   The similarities
233  */
234 TestsCoco.Simulator.Chooser.prototype.similarity = function (tab) {
235     var _this = this,
236         ret = [];
237     
238     $.each(tab, function(index, value) {
239         ret[index] = [];
240         $.each(tab, function(index2, value2) {
241             ret[index][index2] = _this.QuestionSimilarity(index,index2);
242         });
243     });
244     
245     return ret;
246 }

248 /**
249  * Get all the data needed for computing
250  *
251  * @method     getData
252  * @param      {Object}  answers   Set of answers
253  * @param      {Object}  questions   Set of questions
254  */
255 TestsCoco.Simulator.Chooser.prototype.getData = function (answers,questions){
256     
257     //nombre de fois où la question à été vue
258     /**
259     * @prop      {Array} nb_show The number of times the questions were shown 
260     */
261     this.nb_shown = this.countOccurences(answers,questions).shown;
262     
263     //nombre total de vote que la question a reçue
264     /**
265     * @prop      {Array} nb_vote The total number of votes the questions received
266     */
267     this.nb_vote = this.countOccurences(answers,questions).votted;
268     
269     //nombre total de votes positif que la question a reçue
270     /**
271     * @prop      {Array} nb_positive_vote The total number of positive votes the questions received
272     */
273     this.nb_positive_vote = this.countOccurences(answers,questions).positive_vote;
274 
275     //somme de tous les votes
276     /**
277     * @prop      {number} vote_sum The sum of all the votes
278     */
279     this.vote_sum = this.positive(answers);
280 
281     //pourcentage de votes positif par rapport au nombre total de votes
282     /**
283     * @prop      {Array} popularity The percentage of positif votes over the total of votes
284     */
285     this.popularity = this.percentage(this.nb_positive_vote,this.nb_vote);
286 
287     //tableau contenant tous les timecode de début de question
288     /**
289     * @prop      {Array} time The time of all questions
290     */
291     this.time = this.getTime(questions);
292 
293     //tableau contenant tous les énoncés de questions
294     /**
295     * @prop      {Array} enonces The descriptions of all questions
296     */
297     this.enonces = this.getEnonce(questions);
298     
299 };

329 /**
330  * Get the probability of one question
331  *
332  * @method     getProba
333  * @param      {Object}  q          The question
334  * @param      {Object}  tab_score  The scores of the questions
335  * @return     {number}  The probability of the question
336  */
337 TestsCoco.Simulator.Chooser.prototype.getProba = function (q,tab_score){
338     var tool = new TestsCoco.Tools();
339     var array_score = tool.getValuesOfObject(tab_score);
340     var score_sum = array_score.reduce(function(a, b) {
341                                 return a + b;
342                     },0);
343     return this.getScore(q) / score_sum ;
344 }

363 /**
364  * Choose the "best" questions in the set of questions provided in param
365  *
366  * @method     choose
367  * @param      {Object}  answers            Set of answers
368  * @param      {Object}  questions          Set of questions
369  * @param      {number}  numberOfQuestions  The number of questions to choose
370  * @return     {Array}   The choosen questions
371  */
372 TestsCoco.Simulator.Chooser.prototype.choose = function (answers,questions,numberOfQuestions) {
373     var _this = this;
374     var tool = new TestsCoco.Tools();
375     var questionsToDisplay = [];
376     
377     var sim = this.similarity(this.time);
378 
379     var scores = this.getAllScores(questions);
380     
381     var probas = this.getAllProba(questions,scores);
382 
383     var allQuestions = tool.arrayWithProbability(scores);
384     
385     do{
386         var quest = tool.randomWithProbability(allQuestions);
387        
388         if (!_.any(questionsToDisplay,function(value){
389                                                 return sim[quest][value] > this.similarity_threshold;
390                                         })
391         ){
392             questionsToDisplay.push(quest);
393             allQuestions = _.filter(allQuestions,function(id){return id!=quest});
394         }
395         
396     }while(questionsToDisplay.length < numberOfQuestions || !allQuestions)
397     
398     return questionsToDisplay;
399 }

401 /**
402  * Get the choosen questions by the algorithm
403  *
404  * @method     getChoosenQuestions
405  * @param      {Object}  answers                 Set of answers
406  * @param      {Object}  questions               Set of questions
407  * @param      {number}  numberOfQuestions  The number of questions to choose
408  * @return     {Object}  The set of choosen questions
409  */
410 TestsCoco.Simulator.Chooser.prototype.getChoosenQuestions = function (answers,questions,numberOfQuestions){
411 
412     var disp = this.choose(answers,questions,numberOfQuestions);
413 
414     var choosenQuestions = [];
415     
416     $.each(questions, function(index,value){
417         if($.inArray(value.id,disp) != -1){
418             choosenQuestions.push(value);
419         }
420     });
421     
422     return {annotations : choosenQuestions};
423 }

425 /**
426  * The main function of the Chooser class
427  *
428  * @method     main
429  * @param      {Object}  answers            Set of answers
430  * @param      {Object}  questions          Set of questions
431  * @param      {number}  numberOfQuestions  The number of questions to choose
432  * @param      {string}  media              The medias on which we choose the questions
433  * @return     {Object}  The set of questions choosed
434  */
435 TestsCoco.Simulator.Chooser.prototype.main = function (answers,questions,numberOfQuestions,media){
436     var _this = this;
437 
438     var medias = _.groupBy(questions.annotations,'media');
439 
440     this.getData(answers,medias[media]);
441     
442     return this.getChoosenQuestions(answers,medias[media],numberOfQuestions);
443 }
