  8 /**
  9  * Get all stopwords
 10  *
 11  * @method     getStopWords
 12  * @param      {Array}  data1   First set of stopwords
 13  * @param      {Array}  data2   Second set of stopwords
 14  * @return     {Array}  The array of stopwords
 15  */
 16 TestsCoco.Simulator.Questions.prototype.getStopWords = function (data1,data2){
 17     return data1[0].split("\n").concat(data2[0].split("\n")).map(function(s){return s.trim()});
 18 }

 20 /**
 21  * Clean <tt>str</tt>
 22  *
 23  * @method     cleaningText
 24  * @param      {string}  str     The string to clean
 25  * @return     {string}  The cleaned string
 26  */
 27 TestsCoco.Simulator.Questions.prototype.cleaningText = function (str){
 28     return str.toLowerCase()
 29                 .replace(/ÔÅ¨/g," ")
 30                 .replace(/\n/g," ")
 31                 .replace(/[\.,-\/#!$%\^&\*;:{}=\-_`~()\?@""]/g," ")
 32                 .replace(/[\d]/g," ");
 33 }

 35 /**
 36  * Get all the data needed to construct questions
 37  *
 38  * @method     getDonnees
 39  * @param      {Object}  data    Raw input of data
 40  * @return     {Array} The data needed
 41  */
 42 TestsCoco.Simulator.Questions.prototype.getDonnees = function (data){
 43     var _this = this;
 44     var ret = [];
 45     var annotations = data.annotations;
 46     annotations.forEach(function(elem,index){
 47         if(elem.meta["id-ref"] == "Slides"){
 48             var donnee = {};
 49             donnee.deb = elem.begin;
 50             donnee.fin = elem.end;
 51             var txt = elem.content.description;
 52             var mots = _this.cleaningText(txt).split(" ");
 53             donnee.texte = mots ;
 54             ret.push(donnee);
 55         }
 56     });
 57     return ret
 58 }

 60 /**
 61  * Get all the words contained in <tt>data</tt>
 62  *
 63  * @method     getAllWords
 64  * @param      {Object}  data    Set of words
 65  * @return     {Array}  All the words in data
 66  */
 67 TestsCoco.Simulator.Questions.prototype.getAllWords = function (data){
 68     var words = [];
 69     data.forEach(function(value){
 70         words = words.concat(value.texte);
 71     });
 72     return _.uniq(words);
 73 }

 75 /**
 76  * Filter for stopwords
 77  *
 78  * @method     filterStopWords
 79  * @param      {string}  str     The string to filter
 80  * @return     {boolean} 
 81  */
 82 TestsCoco.Simulator.Questions.prototype.filterStopWords = function (str){
 83     return !($.inArray(str,this) > -1);
 84 }
 85 
 86 /**
 87  * Filtering function
 88  *
 89  * @method     filtering
 90  * @param      {Object}  w       Set of words
 91  * @param      {Object}  stpw    Set of stopwords
 92  */
 93 TestsCoco.Simulator.Questions.prototype.filtering = function (w,stpw){
 94     var _this = this;
 95     $.each(w,function(index,value){
 96         var tab_str = value.texte;
 97         var new_tab = tab_str.filter(_this.filterStopWords,stpw);
 98         value.texte=new_tab;
 99     });
100 }

102 TestsCoco.Simulator.Questions.prototype.getFrequencies = function (arr){
103     var ret = {};
104     arr.forEach(function(value){
105         ret[value] = ret[value] ? ret[value]+1 : 1;
106     });

121 /**
122  * Generate random text over a set of words
123  *
124  * @method     generateTxt
125  * @param      {Array}  tab        The set of words
126  * @param      {number}  l_min     Minimal length of the text
127  * @param      {number}  l_max     Maximal length of the text
128  * @param      {Array}  tab_other  Other set of words in wich we can pick words
129  * @return     {string}  The text generated
130  */
131 TestsCoco.Simulator.Questions.prototype.generateTxt = function (tab,l_min,l_max,tab_other){
132     var words;
133     var txt="";
134     var long = _.random(l_min,l_max);
135     
136     if((tab_other != null || tab_other.length != 0 || tab_other != undefined) && (tab.length < long)){
137         words = tab.concat(tab_other);
138     }else{
139         words = tab;
140     }
141     
142     for(var i=0; i<=long; i++){
143             var index = _.random(words.length);
144             txt+= words[index]+" ";
145     }
146     return txt;
147 }

149 /**
150  * Generate one question over a set of words
151  *
152  * @method     generateQuestion
153  * @param      {Array}  tab_mots  Set of words
154  * @param      {number}  longMin   Minimal length of sentences
155  * @param      {number}  longMax   Maximal length of sentences
156  * @param      {number}  nbRepMin  Minimum number of answers
157  * @param      {number}  nbRepMax  Maximum number of answers
158  * @param      {Array}  otherTab  Other set of words in wich we can pick words
159  * @return     {Object}  The question generated
160  */
161 TestsCoco.Simulator.Questions.prototype.generateQuestion = function (tab_mots,longMin,longMax,nbRepMin,nbRepMax,otherTab){
162     var _this = this;
163     var question = {};
164     var enonce = _this.generateTxt(tab_mots,longMin,longMax,otherTab);
165     var reponses = [];
166     var nbRep = _.random(nbRepMin,nbRepMax);
167     var correctRep = _.random(nbRep-1);
168     for(var i = 0; i < nbRep; i++){
169         var ans = {};
170         var rep = _this.generateTxt(tab_mots,longMin,longMax,otherTab);
171         ans.content = rep;
172         if(i == correctRep){
173             ans.correct = true;
174         }
175         reponses[i] = ans;
176     }
177     question.description = enonce;
178     question.answers = reponses;
179     return question;
180 }

182 /**
183  * Generates all the questions for one document
184  *
185  * @method     generate
186  * @param      {Object}  tab         Raw data for generation
187  * @param      {string}  media       The media on wich we generate the questions
188  * @param      {number}  nombre      The number of questions to generate
189  * @param      {number}  longMin     The minimum length of sentences
190  * @param      {number}  longMax     The maximum length of sentences
191  * @param      {number}  nbRepMin    The minimum number of answers by questions
192  * @param      {number}  nbRepMax    The maximum number of answers by questions
193  * @param      {Array}  otherWords   Other set of words in wich we can pick words
194  * @return     {Array}   The questions generated
195  */
196 TestsCoco.Simulator.Questions.prototype.generate = function (tab,media,nombre,longMin,longMax,nbRepMin,nbRepMax,otherWords){
197     var tool = new TestsCoco.Tools();
198     var _this = this;
199     var retour=[];
200     for(var i = 0 ; i < nombre ; i++){
201        var idx = _.random(tab.length-1);
202        var obj = tab[idx];
203        var obj_av = tab[idx-1];
204        var obj_ap = tab[idx+1];
205        var time = _this.generateTime(obj);
206        var mots; 
207        var t1 = obj.texte, t2, t3;
208        if(idx > 0 && idx < tab.length - 1){
209            t2 = tab[idx-1].texte;
210            t3 = tab[idx+1].texte;
211            mots = t2.concat(t1).concat(t3);
212        }else if (idx == 0){
213            t2 = tab[idx+1].texte;
214            mots = t1.concat(t2);
215        }else{
216            t2=tab[idx-1].texte;
217            mots = t1.concat(t2);
218        }
219        var quest= {};
220        quest.content =  _this.generateQuestion(mots,longMin,longMax,nbRepMin,nbRepMax,otherWords);
221        quest.begin = time;
222        quest.end = time+3000;
223        quest.type = "Quizz";
224        quest.media = media;
225        quest.id = tool.generateUid();
226        retour[i] = quest;
227     }
228 
229     return retour;
230 }

232 /**
233  * The main function of the Questions class
234  *
235  * @method     main
236  * @param      {Object}  stop_word1    First set of stopwords
237  * @param      {Object}  stop_word2    Second set of stopwords
238  * @param      {Array}  documents     All the documents on wich we generate questions
239  * @param      {boolean}  other        Tell if we get words in other slides
240  * @param      {number}  nb_questions  The number of question to generate in each document
241  * @return     {Object}  The questions generated
242  */
243 TestsCoco.Simulator.Questions.prototype.main = function (stop_word1,stop_word2,documents,other,nb_questions){
244     var _this = this;
245     var stopwords_fr = this.getStopWords(stop_word1,stop_word2);
246     var allQuest = [];
247     documents.forEach(function(elem){
248 
249         var media = elem.medias[0].id;
250         var donnees = _this.getDonnees(elem);
251 
252         _this.filtering(donnees,stopwords_fr);
253         
254         var all_words = other ? _this.getAllWords(donnees) : [];
255         
256         allQuest = allQuest.concat(_this.generate(donnees,media,nb_questions,5,10,2,5,all_words));
257     });
258     
259     return {"annotations" : allQuest};
260     
261 }
