 15 TestsCoco.Simulator.Questions.prototype.getDonnees = function (data){
 16     var _this = this;
 17     var ret = [];
 18     var annotations = data.annotations;
 19     annotations.forEach(function(elem,index){
 20         if(elem.meta["id-ref"] == "Slides"){
 21             var donnee = {};
 22             donnee.deb = elem.begin;
 23             donnee.fin = elem.end;
 24             var txt = elem.content.description;
 25             var mots = _this.cleaningText(txt).split(" ");
 26             donnee.texte = mots ;
 27             ret.push(donnee);
 28         }
 29     });
 30     return ret

 33 TestsCoco.Simulator.Questions.prototype.getAllWords = function (data){

 66 TestsCoco.Simulator.Questions.prototype.generateTxt = function (tab,l_min,l_max,tab_other){
 67     var words;
 68     var txt="";
 69     var long = _.random(l_min,l_max);
 70     
 71     if((tab_other != null || tab_other.length != 0 || tab_other != undefined) && (tab.length < long)){
 72         words = tab.concat(tab_other);
 73     }else{
 74         words = tab;
 75     }
 76     
 77     for(var i=0; i<=long; i++){
 78             var index = _.random(words.length);
 79             txt+= words[index]+" ";
 80     }
 81     return txt;
 82 }

 84 TestsCoco.Simulator.Questions.prototype.generateQuestion = function (tab_mots,longMin,longMax,nbRepMin,nbRepMax,otherTab){
 85     var _this = this;
 86     var question = {};
 87     var enonce = _this.generateTxt(tab_mots,longMin,longMax,otherTab);
 88     var reponses = [];
 89     var nbRep = _.random(nbRepMin,nbRepMax);
 90     var correctRep = _.random(nbRep-1);
 91     for(var i = 0; i < nbRep; i++){
 92         var ans = {};
 93         var rep = _this.generateTxt(tab_mots,longMin,longMax,otherTab);
 94         ans.content = rep;
 95         if(i == correctRep){
 96             ans.correct = true;
 97         }
 98         reponses[i] = ans;
 99     }
100     question.description = enonce;
101     question.answers = reponses;
102     return question;
103 }

105 TestsCoco.Simulator.Questions.prototype.generate = function (tab,media,nombre,longMin,longMax,nbRepMin,nbRepMax,otherWords){
106     var tool = new TestsCoco.Tools();
107     var _this = this;
108     var retour=[];
109     for(var i = 0 ; i < nombre ; i++){
110        var idx = _.random(tab.length-1);
111        var obj = tab[idx];
112        var obj_av = tab[idx-1];
113        var obj_ap = tab[idx+1];
114        var time = _this.generateTime(obj);
115        var mots; 
116        var t1 = obj.texte, t2, t3;
117        if(idx > 0 && idx < tab.length - 1){
118            t2 = tab[idx-1].texte;
119            t3 = tab[idx+1].texte;
120            mots = t2.concat(t1).concat(t3);
121        }else if (idx == 0){
122            t2 = tab[idx+1].texte;
123            mots = t1.concat(t2);
124        }else{
125            t2=tab[idx-1].texte;
126            mots = t1.concat(t2);
127        }
128        var quest= {};
129        quest.content =  _this.generateQuestion(mots,longMin,longMax,nbRepMin,nbRepMax,otherWords);
130        quest.begin = time;
131        quest.end = time+3000;
132        quest.type = "Quizz";
133        quest.media = media;
134        quest.id = tool.generateUid();
135        retour[i] = quest;
136     }
137 
138     return retour;
139 }

141 TestsCoco.Simulator.Questions.prototype.main = function (stop_word1,stop_word2,documents,other,nb_questions){
142     var _this = this;
143     var stopwords_fr = this.getStopWords(stop_word1,stop_word2);
144     var allQuest = [];
145     documents.forEach(function(elem){
146 
147         var media = elem.medias[0].id;
148         var donnees = _this.getDonnees(elem);
149 
150         _this.filtering(donnees,stopwords_fr);
151         
152         var all_words = other ? _this.getAllWords(donnees) : [];
153         
154         allQuest = allQuest.concat(_this.generate(donnees,media,nb_questions,5,10,2,5,all_words));
155     });
156     
157     return {"annotations" : allQuest};
158     
159 }
