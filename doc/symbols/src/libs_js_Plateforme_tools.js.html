 26 TestsCoco.Tools.prototype.arrayUnique = function (array) {
 27     var a = array.concat();
 28     for(var i=0; i<a.length; ++i) {
 29         for(var j=i+1; j<a.length; ++j) {
 30             if(a[i] === a[j])
 31                 a.splice(j--, 1);
 32         }
 33     }
 34 
 35     return a;
 36 }

 67 /**
 68  * Make the transposition of an array
 69  *
 70  * @method     transpose
 71  * @param      {Array}  a      
 72  * @return     {Array}  The tansposed array
 73  */
 74 TestsCoco.Tools.prototype.transpose = function(a) {
 75 
 76     // Calculate the width and height of the Array
 77     var w = a.length ? a.length : 0,
 78         h = a[0] instanceof Array ? a[0].length : 0;
 79 
 80     // In case it is a zero matrix, no transpose routine needed.
 81     if(h === 0 || w === 0) { return []; }
 82 
 83     /**
 84     * @var {Number} i Counter
 85     * @var {Number} j Counter
 86     * @var {Array} t Transposed data is stored in this array.
 87     */
 88     var i, j, t = [];
 89 
 90     // Loop through every item in the outer array (height)
 91     for(i=0; i<h; i++) {
 92 
 93         // Insert a new row (array)
 94         t[i] = [];
 95 
 96         // Loop through every item per item in outer array (width)
 97         for(j=0; j<w; j++) {
 98 
 99           // Save transposed data.
100           t[i][j] = a[j][i];
101         }
102     }
103 
104     return t;
105 };

107 /**
108  * Make the dot product of 2 vectors
109  *
110  * @method     dot
111  * @param      {number[]}  v1      The first vector
112  * @param      {number[]}  v2      The second vector
113  * @return     {number}  The dot product
114  */
115 TestsCoco.Tools.prototype.dot = function (v1,v2){
116     var res = 0;
117     for(var i = 0;i < v1.length;i++){
118         res+= (v1[i]*v2[i]);
119     }
120     return res;
121 }

146 /**
147  * Generate an array with the number of occurence of each value of <tt>tab</tt>
148  *
149  * @method     arrayWithProbability
150  * @param      {Object}  tab     The array of probabilities
151  * @return     {Array}   The array generated
152  * @example    //Example
153  * var data = {1:0.4,
154  *             2:0.1,
155  *             3:0.2,
156  *             4:0.3}; 
157  * var arr = arrayWithProbability(data); 
158  * console.log(arr); ( --> : [1,1,1,1,2,3,3,4,4,4])
159  */
160 TestsCoco.Tools.prototype.arrayWithProbability = function (tab) {
161     var arr = [];
162     $.each(tab, function(index,value){
163         for(var i = 0; i < value; i++){
164             arr.push(index);
165         }
166     });
167     return arr;
168 }

183 /**
184  * Get the value of the query variable passed in the url
185  *
186  * @method     getQueryVariable
187  * @param      {string}  variable  The variable searched
188  * @return     {string}  The value of the variable
189  */
190 TestsCoco.Tools.prototype.getQueryVariable = function (variable) {
191     var query = window.location.search.substring(1);
192     var vars = query.split('&');
193     for (var i = 0; i < vars.length; i++) {
194         var pair = vars[i].split('=');
195         if (decodeURIComponent(pair[0]) == variable) {
196             return decodeURIComponent(pair[1]);
197         }
198     }
199     console.log('Query variable %s not found', variable);
200 }
