 24 TestsCoco.Tools.prototype.arrayUnique = function (array) {
 25     var a = array.concat();
 26     for(var i=0; i<a.length; ++i) {
 27         for(var j=i+1; j<a.length; ++j) {
 28             if(a[i] === a[j])
 29                 a.splice(j--, 1);
 30         }
 31     }
 32 
 33     return a;
 34 }

 76 /**
 77  * Make the transposition of an array
 78  *
 79  * @method     transpose
 80  * @param      {Array}  a      
 81  * @return     {Array}  The tansposed array
 82  */
 83 TestsCoco.Tools.prototype.transpose = function(a) {
 84 
 85     // Calculate the width and height of the Array
 86     var w = a.length ? a.length : 0,
 87         h = a[0] instanceof Array ? a[0].length : 0;
 88 
 89     // In case it is a zero matrix, no transpose routine needed.
 90     if(h === 0 || w === 0) { return []; }
 91 
 92     /**
 93     * @var {Number} i Counter
 94     * @var {Number} j Counter
 95     * @var {Array} t Transposed data is stored in this array.
 96     */
 97     var i, j, t = [];
 98 
 99     // Loop through every item in the outer array (height)
100     for(i=0; i<h; i++) {
101 
102         // Insert a new row (array)
103         t[i] = [];
104 
105         // Loop through every item per item in outer array (width)
106         for(j=0; j<w; j++) {
107 
108           // Save transposed data.
109           t[i][j] = a[j][i];
110         }
111     }
112 
113     return t;
114 };

116 /**
117  * Make the dot product of 2 vectors
118  *
119  * @method     dot
120  * @param      {number[]}  v1      The first vector
121  * @param      {number[]}  v2      The second vector
122  * @return     {number}  The dot product
123  */
124 TestsCoco.Tools.prototype.dot = function (v1,v2){
125     var res = 0;
126     for(var i = 0;i < v1.length;i++){
127         res+= (v1[i]*v2[i]);
128     }
129     return res;
130 }

155 /**
156  * Generate an array with the number of occurence of each value of <tt>tab</tt>
157  *
158  * @method     arrayWithProbability
159  * @param      {Object}  tab     The array of probabilities
160  * @return     {Array}   The array generated
161  * @example    //Example
162  * var data = {1:0.4,
163  *             2:0.1,
164  *             3:0.2,
165  *             4:0.3}; 
166  * var arr = arrayWithProbability(data); 
167  * console.log(arr); ( --> : [1,1,1,1,2,3,3,4,4,4])
168  */
169 TestsCoco.Tools.prototype.arrayWithProbability = function (tab) {
170     var arr = [];
171     $.each(tab, function(index,value){
172         for(var i = 0; i < value; i++){
173             arr.push(index);
174         }
175     });
176     return arr;
177 }

192 /**
193  * Get the value of the query variable passed in the url
194  *
195  * @method     getQueryVariable
196  * @param      {string}  variable  The variable searched
197  * @return     {string}  The value of the variable
198  */
199 TestsCoco.Tools.prototype.getQueryVariable = function (variable) {
200     var query = window.location.search.substring(1);
201     var vars = query.split('&');
202     for (var i = 0; i < vars.length; i++) {
203         var pair = vars[i].split('=');
204         if (decodeURIComponent(pair[0]) == variable) {
205             return decodeURIComponent(pair[1]);
206         }
207     }
208     console.log('Query variable %s not found', variable);
209 }

211 /**
212  * Generate visualisation by types
213  *
214  * @method     visualize
215  * @param      {DataVis}  visualizer  Instance of DataVis module
216  * @param      {string}  type        Type of visualisation (student,teacher or algorithm visualisation)
217  */
218 TestsCoco.Tools.prototype.visualize = function (visualizer,type) {
219 
220     var _this = this;
221     function checkType(){
222         if(type == 'student'){
223             var user = _this.getQueryVariable('username');
224             var session = _this.getQueryVariable('session');
225             visualizer.generateGraphStudent(user,session);
226         }else if (type == 'teacher'){
227             var media = _this.getQueryVariable('mediaId');
228             visualizer.generateGraphTeacher(media);
229         }else {
230             visualizer.generateGraphVisuAlgo();
231         }
232     }
233 
234     var questions_filepath = "../Donnees_tests/analytics_data/questions_3files.json";
235     var answers_filepath = "../Donnees_tests/analytics_data/answers_3files.json";
236 
237     if(localStorage.getItem("sim_question") !== null && localStorage.getItem("sim_answer") !== null){
238         var local_question = localStorage.getItem('sim_question');
239         var local_answer = localStorage.getItem('sim_answer');
240         visualizer.getAllData(JSON.parse(LZString.decompress(local_question)),JSON.parse(LZString.decompress(local_answer)));
241         checkType();
242     }else{
243         $.when($.get(questions_filepath),
244                     $.get(answers_filepath))
245                 .done(function(questions,answers){
246                     visualizer.getAllData(questions[0],answers[0]);
247                     checkType();
248                 });
249     }
250 }
