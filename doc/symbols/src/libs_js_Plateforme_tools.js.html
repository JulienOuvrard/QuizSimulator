 19 TestsCoco.Tools.prototype.arrayUnique = function (array) {
 20     var a = array.concat();
 21     for(var i=0; i<a.length; ++i) {
 22         for(var j=i+1; j<a.length; ++j) {
 23             if(a[i] === a[j])
 24                 a.splice(j--, 1);
 25         }
 26     }
 27 
 28     return a;
 29 }

 43 TestsCoco.Tools.prototype.transpose = function(a) {
 44 
 45       // Calculate the width and height of the Array
 46       var w = a.length ? a.length : 0,
 47         h = a[0] instanceof Array ? a[0].length : 0;
 48 
 49       // In case it is a zero matrix, no transpose routine needed.
 50       if(h === 0 || w === 0) { return []; }
 51 
 52       /**
 53        * @var {Number} i Counter
 54        * @var {Number} j Counter
 55        * @var {Array} t Transposed data is stored in this array.
 56        */
 57       var i, j, t = [];
 58 
 59       // Loop through every item in the outer array (height)
 60       for(i=0; i<h; i++) {
 61 
 62         // Insert a new row (array)
 63         t[i] = [];
 64 
 65         // Loop through every item per item in outer array (width)
 66         for(j=0; j<w; j++) {
 67 
 68           // Save transposed data.
 69           t[i][j] = a[j][i];
 70         }
 71       }
 72 
 73       return t;
 74 };

 76 TestsCoco.Tools.prototype.dot = function (v1,v2){
 77     var res = 0;
 78     for(var i = 0;i < v1.length;i++){
 79         res+= (v1[i]*v2[i]);
 80     }
 81     return res;
 82 }

116 TestsCoco.Tools.prototype.getQueryVariable = function (variable) {
117     var query = window.location.search.substring(1);
118     var vars = query.split('&');
119     for (var i = 0; i < vars.length; i++) {
120         var pair = vars[i].split('=');
121         if (decodeURIComponent(pair[0]) == variable) {
122             return decodeURIComponent(pair[1]);
123         }
124     }
125     console.log('Query variable %s not found', variable);
126 }
