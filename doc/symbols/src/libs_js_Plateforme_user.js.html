1 /**
  2  * Constructor of the User class
  3  *
  4  * @class     <type> (User)
  5  *
  6  * @param      {String}  name           { description }
  7  * @param      {String}  profile        { description }
  8  * @param      {Date[]}  session_dates  { description }
  9  */
 10 TestsCoco.User = function(name,profile,session_dates){
 11     this.name = name;
 12     this.profile = profile;
 13     this.session_dates = session_dates;
 14     this.bias = {
 15         answer_rate : 0.7,
 16         vote_rate : 0.9,
 17         right_answer_rate : 0.8,
 18         usefull_vote_rate : 0.7
 19     };
 20     this.date_property = {};
 21     this.date_property.dayInMillisecond = 86400000;
 22     this.date_property.hourInMillisecond = 3600000;
 23     this.date_property.minuteInMillisecond = 60000;
 24 }
 25 
 26 /**
 27  * Setter for the session date of the user
 28  *
 29  * @class     <type> (User)
 30  *
 31  * @param      {Date}  start      { The date when all the sessions starts }
 32  * @param      {number}  nb_tours   { The number of sessions to generate }
 33  * @param      {Array}   documents  { The documents on which the sessions are made }
 34  *
 35  * @return     {Array}   { The sessions date of the user }
 36  */
 37 TestsCoco.User.prototype.setSessionDates = function(start,nb_tours,documents) {
 38     var _this = this;
 39     var tool = new TestsCoco.Tools();
 40     var dates = [];
 41     
 42     var cases = {
 43       "regular": function(tab) {
 44                     var doc = _.random(documents.length-1);
 45                     var media = documents[doc].medias[0].id;
 46                     var session_id = tool.generateUid();
 47                     tab.push({'id':session_id,'date':start,'media':media});
 48                     for(var i = 1; i < nb_tours; i++){
 49                         var session_id = tool.generateUid();
 50                         var doc = _.random(documents.length-1);
 51                         var media = documents[doc].medias[0].id;
 52                         var max_time = _.max(documents[doc].annotations,"end").end;
 53                         var d = new Date();
 54                         var offset = d.getTimezoneOffset();
 55                         var temp = tab[i-1].date;
 56                         var hours = _.random(9,20);
 57                         d.setTime(temp.getTime() + offset + _.random(1) * _this.date_property.dayInMillisecond + max_time + hours * _this.date_property.hourInMillisecond);
 58                         tab.push({'id':session_id,'date':d,'media':media});
 59                     }
 60                     return tab;
 61                 },
 62       "random": function(tab) {
 63                     var doc = _.random(documents.length-1);
 64                     var media = documents[doc].medias[0].id;
 65                     var session_id = tool.generateUid();
 66                     tab.push({'id':session_id,'date':start,'media':media});
 67                     for(var i = 1; i < nb_tours; i++){
 68                         var session_id = tool.generateUid();
 69                         var doc = _.random(documents.length-1);
 70                         var media = documents[doc].medias[0].id;
 71                         var max_time = _.max(documents[doc].annotations,"end").end;
 72                         var d = new Date();
 73                         var offset = d.getTimezoneOffset();
 74                         var temp = tab[i-1].date;
 75                         var hours = _.random(9,20);
 76                         d.setTime(temp.getTime() + offset + _.random(3) * _this.date_property.dayInMillisecond + max_time + hours * _this.date_property.hourInMillisecond);
 77                         tab.push({'id':session_id,'date':d,'media':media});
 78                     }
 79                     return tab;
 80                 },
 81       _default: function() { console.log('Profile not implemented'); }
 82     };
 83 
 84     this.session_dates = cases[ this.profile ] ? cases[ this.profile ](dates) : cases._default();
 85 }
